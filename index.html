			function observable() {
			  return 1;
			  setTimeout(() => {
			    return 42;
			  }, 1000);
			}

 

			function observable(observer) {
			  observer.onNext(1);
			  setTimeout(() => {
			    observer.onNext(42);
			  }, 1000);
			}

 

			observable({
			  onNext: value => console.log(`Next value: ${value}`);
			});
			// Next value: 1
			// Next value: 42

 

			function observable(observer) {
			  let i = 0;
			  const id = setInterval(() => {
			    observer.onNext(++i);
			  }, 1000);
			  return () => clearInterval(id);
			}

 

			const unsubscribe = observable({
			  onNext: value => console.log(`Next value: ${value}`);
			});
			// Next value: 1
			// Next value: 2
			// и тд

 

			function filter(observable, fn) {
			  return function(observer) {
			    return observable({
			      onNext: value => fn(value) && observer.next(value)
			    });
			  }
			}

 

			const oddNumbers = filter(observable, v => v % 2 === true);
			const unsubscribe = oddNumbers({
			  onNext: value => console.log(`Next value: ${value}`);
			});
			// Next value: 1
			// Next value: 3
			// и тд

 

			function map(observable, fn) {
			  return function(observer) {
			    return observable({
			      onNext: value => observer.next(fn(value))
			    });
			  }
			}

 

			const bigNumbers = map(observable, v => v * 100);
			const unsubscribe = bigNumbers({
			  onNext: value => console.log(`Next value: ${value}`);
			});
			// Next value: 100
			// Next value: 200
			// и тд

 

			const oddNumbers = filter(observable, v => v % 2 === true)
			const bigOddNumbers = map(oddNumbers, v => v * 100);
			const unsubscribe = bigNumbers({
			  onNext: value => console.log(`Next value: ${value}`);
			});
			// Next value: 100
			// Next value: 300
